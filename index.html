async function load() { const container = document.getElementById('catalog'); try { // Додаємо випадкове число до запиту, щоб iPhone не брав стару версію з пам'яті const r = await fetch(SHEET_CSV + '&nocache=' + Math.random()); const data = await r.text(); const rows = data.split('\n').slice(1).map(line => line.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/) .map(v => v.replace(/^"|"$/g, '').trim()) ).filter(c => c.length >= 2); container.innerHTML = ''; rows.forEach((c, index) => { let [brand, name, priceStr, minStr, desc, img] = c; if (!brand || !name) return; const price = parseFloat(priceStr.replace(',', '.')) || 0; const min = parseInt(minStr) || 2; // Очищення тексту від усіх небезпечних символів const cleanBrand = brand.replace(/['"/]/g, '’'); const cleanName = name.replace(/['"/]/g, '’'); const cleanDesc = desc ? desc.replace(/['"/]/g, '’') : ""; const fullTitle = `${cleanBrand} ${cleanName}`; let opts = ''; [2, 5, 10].forEach(ml => { if (ml >= min) { const totalPrice = Math.round(ml * price); // Створюємо готове повідомлення для кожної опції const msg = encodeURIComponent(`🛍️ ЗАМОВЛЕННЯ:\n${fullTitle}\nОб'єм: ${ml} мл\nСума: ${totalPrice} грн`); const tgLink = `https://t.me/viktoriia_viktoriia_r?text=${msg}`; opts += ``; } }); const div = document.createElement('div'); div.className = 'card'; div.innerHTML = `
${cleanBrand}
${cleanName}
1 мл: ${Math.round(price)} грн
${Math.round(min * price)} грн
ЗАМОВИТИ `; container.appendChild(div); // Ініціалізуємо перше посилання для кнопки updateBtn(div.querySelector('select')); }); } catch (e) { container.innerText = "Помилка завантаження."; } } // Функція, яка миттєво міняє посилання в кнопці при виборі об'єму function updateBtn(select) { const link = select.value; // Беремо готовий лінк з option const btn = select.nextElementSibling.nextElementSibling; const priceText = select.options[select.selectedIndex].text.split(' — ')[1]; btn.href = link; select.nextElementSibling.innerText = priceText; }
